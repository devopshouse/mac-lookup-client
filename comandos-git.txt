# Instalação do cliente git
- MacOS: brew install git
- Linux debian based: apt-get install git
- Linux redhat based: yum install git
- Windows: Baixar o isntalador MSI

# Inicializar repositorio git local
git init

# Status do repositorio
git status

# Adicionar arquivos na area de stag
git add NOME_DO_ARQUIVO

# Commitar arquivos na area de stag
git commit -m "MENSAGEM AMIGAVEL"

# Mostrar histórico de commits
git log
